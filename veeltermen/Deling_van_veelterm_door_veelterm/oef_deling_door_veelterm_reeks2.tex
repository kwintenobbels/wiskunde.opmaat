\begin{exercise}[\bf \ref{antw2.7}.]\setcounter{enumi}{7}

		\begin{question} $A(x) = x^2-x+6$                              \quad \text{en}           \quad $B(x) = x+2$                   \( \answer[onlineshowanswerbutton]{  } \) \end{question}
		\begin{question} $A(x) = x^3+x^2-9x+1$                         \quad \text{en}           \quad $B(x) = x^2+1$                 \( \answer[onlineshowanswerbutton]{  } \) \end{question}
		\begin{question} $A(x) = 2x^3 +5x^2+9x+9$                      \quad \text{en}           \quad $B(x) = x^2 + x + 3$           \( \answer[onlineshowanswerbutton]{  } \) \end{question}
		\begin{question} $A(x) = x^4 + x^3 + 5x^2 + 4x + 4$            \quad \text{en}           \quad $B(x) = x^2 + x + 1$           \( \answer[onlineshowanswerbutton]{  } \) \end{question}
		\begin{question} $A(x) = 4x^4 + 7x^2 + 13x - 5$                \quad \text{en}           \quad $B(x) = 2x^2 + x + 1$          \( \answer[onlineshowanswerbutton]{  } \) \end{question}

\begin{exercise}[\bf \ref{antw2.8}.]\setcounter{enumi}{8}  
\hypertarget{oef2.8}{Toon} telkens aan dat de veelterm $A(x)$ deelbaar is door de veelterm $B(x)$ en schrijf $A(x)$ als een product van twee veeltermen.

		\begin{question} $A(x) = x^4 + x^2 + 1$                        \quad \text{en}           \quad $B(x) = x^2 + x + 1$           \( \answer[onlineshowanswerbutton]{  } \) \end{question}
		\begin{question} $A(x) = x^5 - 1$                              \quad \text{en}           \quad $B(x) = x - 1$                 \( \answer[onlineshowanswerbutton]{  } \) \end{question}
		\begin{question} $A(x) = x^4 + 1$                              \quad \text{en}           \quad $B(x) = x^2 + \sqrt{2}\,x + 1$ \( \answer[onlineshowanswerbutton]{  } \) \end{question}

\end{exercise}

\begin{exercise}[\bf \ref{antw2.9}.]\setcounter{enumi}{9} 
\hypertarget{oef2.9}{Gegeven} zijn de volgende veeltermen waarbij de letters $a$, $b$ en $c$ re\"ele getallen voorstellen. Bepaal, door middel van de methode van de onbepaalde co\"effici\"enten, telkens de waarde(n) van de parameters $a$, $b$ en $c$ opdat $A(x)$ deelbaar zou zijn door $B(x)$. 

		\begin{question} $A(x) = x^3 + ax^2 + bx + 1$                  \quad \text{ en } \quad $B(x) = x^2 + x + 1$                   \( \answer[onlineshowanswerbutton]{  } \) \end{question}
		\begin{question} $A(x) = 3x^3 - ax^2 - 4x + 8$                 \quad \text{ en } \quad $B(x) = x^2 + bx + 4$                  \( \answer[onlineshowanswerbutton]{  } \) \end{question}
		\begin{question} $A(x) = 4x^4 + ax^3 + 7x^2 + bx + 3$          \quad \text{ en } \quad $B(x) = 2x^2 + x + 1$                  \( \answer[onlineshowanswerbutton]{  } \) \end{question}
		\begin{question} $A(x) = 3x^4 + ax^3 -ax - 35$                 \quad \text{ en } \quad $B(x) = x^3 + bx^2 + cx + 7$           \( \answer[onlineshowanswerbutton]{  } \) \end{question}
		\begin{question} $A(x) = x^4+1$                                \quad \text{ en } \quad $B(x) = x^2+ax+b$                      \( \answer[onlineshowanswerbutton]{  } \) \end{question}
		\begin{question} $A(x) = x^4+x^2+1$                            \quad \text{ en } \quad $B(x) = x^2+ax+b$                      \( \answer[onlineshowanswerbutton]{  } \) \end{question}

\end{exercise}

\begin{exercise}[\bf \ref{antw2.10}.]\setcounter{enumi}{10} 
\hypertarget{oef2.10}{Bepaal} telkens het quoti\"ent en de rest bij deling van $A(x)$ door $B(x)$. Schrijf ook telkens het verband op tussen deeltal, deler, quoti\"ent en rest. Geef ook aan of $A(x)$ deelbaar is door $B(x)$.

		\begin{question} $A(x) = x^2-1$                                \quad \text{en}           \quad $B(x) = x+1$                   \( \answer[onlineshowanswerbutton]{  } \) \end{question}
		\begin{question} $A(x) = 2x-8$                                 \quad \text{en}           \quad $B(x) = -6$                    \( \answer[onlineshowanswerbutton]{  } \) \end{question}
		\begin{question} $A(x) = x^3 +125$                             \quad \text{en}           \quad $B(x) = x^2 - 5x + 25$         \( \answer[onlineshowanswerbutton]{  } \) \end{question}
		\begin{question} $A(x) = 3x^3 + 5x^2 + 7x + 9$                 \quad \text{en}           \quad $B(x) = x^2 + x + 1$           \( \answer[onlineshowanswerbutton]{  } \) \end{question}
		\begin{question} $A(x) = x^3 + 3x^2 + 5x + 1$                  \quad \text{en}           \quad $B(x) = x^2 + 4x + 2$          \( \answer[onlineshowanswerbutton]{  } \) \end{question}
$A(x) = x^2+x+1$ en $B(x) = 3x-2$ 
$A(x) = 2x^3 + x^2 + 1$ en $B(x) = 4x^2 + 5$ 

\end{exercise}
