\begin{exercise}[\bf \ref{antw3.2}.]\setcounter{enumi}{2}

\begin{xmmulticols}{4}
	\begin{enumerate}
		\item $a^2 $
		\item $b-a $
		\item $a-b $
		\item $b   $
	\end{enumerate}

\end{xmmulticols}
\end{exercise}
